CREATE DATABASE IF NOT EXISTS eduinfo;

use eduinfo;

CREATE TABLE IF NOT EXISTS schools (
id INT AUTO_INCREMENT PRIMARY KEY,
voivodeship VARCHAR(50),
county VARCHAR(50),
community VARCHAR(50),
type VARCHAR(255),
name VARCHAR(255),
street VARCHAR(50),
houseNumber VARCHAR(10),
postalCode VARCHAR(50),
phoneNumber VARCHAR(50),
website VARCHAR(255),
students INT,
branches INT,
email VARCHAR(255),
town VARCHAR(255),
lat DOUBLE,
lon DOUBLE);

LOAD DATA LOCAL INFILE "<<PATH TO schools.csv>>"
INTO TABLE schools
FIELDS TERMINATED BY ','
ENCLOSED BY '"' LINES
TERMINATED BY '\n';
